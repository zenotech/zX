# Kriging surrogate + gradient descent exploration from pandas import DataFrame from sklearn.gaussian_process import GaussianProcessRegressor from sklearn.gaussian_process.kernels import Matern from scipy.optimize import minimize import numpy as np def explore(table: DataFrame, state: dict) -> list[dict]: """Bayesian optimization via Kriging surrogate.""" max_iter = state.get("max_iterations", 5) iteration = state.get("iteration", 0) if iteration >= max_iter: return [] done = table[table["_zx_status"] == "completed"] X = done[["x1", "x2"]].values y = done["f_x"].values gpr = GaussianProcessRegressor(kernel=Matern(nu=2.5)) gpr.fit(X, y) x0 = X[np.argmin(y)] result = minimize( lambda x: gpr.predict([x])[0], x0, method="L-BFGS-B", bounds=[(-3, 3), (-2, 2)] ) state["iteration"] = iteration + 1 return [{"x1": result.x[0], "x2": result.x[1]}]
